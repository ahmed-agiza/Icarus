CREATE TABLE `Pictures` (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`PictureID`	TEXT NOT NULL,
	`Path_pic`	TEXT NOT NULL,
	`OwnerID`	TEXT NOT NULL,
	FOREIGN KEY(`OwnerID`) REFERENCES Peers(UID)
);
CREATE TABLE "Permissions" (
	`TransID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`PictureID`	TEXT NOT NULL,
	`UID`	TEXT NOT NULL,
	`RCount`	INTEGER NOT NULL,
	FOREIGN KEY(`PictureID`) REFERENCES Pictures(PictureID),
	FOREIGN KEY(`UID`) REFERENCES Peers(UID)
);
CREATE TABLE "Peers" (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`UID`	TEXT NOT NULL,
	`Username`	TEXT NOT NULL,
	`PublicKey`	TEXT NOT NULL,
	`Key2`	TEXT NOT NULL
);
CREATE TABLE "Requests" (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`RequestID`	TEXT NOT NULL,
	`OwnerID`	TEXT NOT NULL,
	`SenderID`	TEXT NOT NULL,
	`Timestamp` BLOB NOT NULL,
	FOREIGN KEY(`OwnerID`) REFERENCES Peers(UID),
	FOREIGN KEY(`SenderID`) REFERENCES Peers(UID)
);
